#+begin_src C

  #include<stdlib.h>
  #include<unistd.h>
  #include<fcntl.h>

  int main(){

    char *strs[7] = {"a", "b", "c", "d", "e", "f", "g"};
    int fd = open("testfile", O_CREAT|O_RDWR);

    for(int i=0; i<7; i++){
      write(fd, strs[i], 1);
    }
    fsync(fd);
    // =========> Persistence guarantee
    close(fd);
  }

#+end_src
